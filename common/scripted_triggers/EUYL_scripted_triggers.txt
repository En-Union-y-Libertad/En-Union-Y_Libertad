PBC_check_for_war_focus = {
	if = {
		limit = {
			COL = { 
				has_completed_focus = COL_revivir_la_confederacion
			}
		}
		OR = {
			has_civil_war = yes 
			has_war_with = BOL 
			has_war_with = PRU
		}
	}
	else = {
		OR = {
			has_war_with = CHL 
			has_war_with = ARG 
			has_war_with = ECU 
			has_war_with = COL 
		}
	}
}
