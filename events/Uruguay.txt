add_namespace = uruguay

#event that will trigger

country_event = {
	id = uruguay.1
	title = uruguay.1.t
	desc = uruguay.1.d
	
	hidden = yes
	
	trigger = {
		has_country_flag = URG_terra
		check_variable = { URG_politicians_arrested > 4 }
		date > 1936.2.1
	}
	
	mean_time_to_happen = { days = 85 }
	
	immediate = {
		random_list = {
			50 = {
				country_event = { id = uruguay.2 days = 25 }
			}
			50 = {
				country_event = { id = uruguay.3 days = 25 }
			}
		}
	}
}

#Uruguay gets notification of Argentina accepting the pact

country_event = {
	id = uruguay.9
	title = uruguay.9.t 
	desc = uruguay.9.d 
	
	is_triggered_only = yes 
	
	option = {
		name = uruguay.9.a 
		if = {
			limit = { 
				URG = {
					is_in_faction = yes 
					is_faction_leader = yes 
				}
			}
			URG = {
				add_to_faction = ARG 
			}
			hidden_effect = {
				URG = {
					set_country_flag = URG_ARG_agreement
				}
				ARG = {
					set_country_flag = URG_ARG_agreement
				}
			}
		}
		else_if = {
			limit = {
				ARG = {
					is_in_faction = yes 
					is_faction_leader = yes 
				}
			}
			ARG = {
				add_to_faction = URG 
			}
			hidden_effect = {
				URG = {
					set_country_flag = URG_ARG_agreement
				}
				ARG = {
					set_country_flag = URG_ARG_agreement
				}
			}
		}
		else = {
			diplomatic_relation = {
				country = ARG 
				relation = non_aggression_pact
			}
			hidden_effect = {
				URG = {
					set_country_flag = URG_ARG_agreement
				}
				ARG = {
					set_country_flag = URG_ARG_agreement
				}
			}
		}		
	}
}
